Example:

Code:

            var Model = Backbone.Model.extend({
                schema: {
                    name: {
                        type: 'TypeAhead',
                        title: 'Name',
                        options: ['val1', 'val2', 'val3', 'example', 'test']
                    }
                }
            });
            var model = new Model({name: 'val1'});
            var form = new Backbone.Form({
                model: model
            });
            $('#form').html(form.render().$el);
